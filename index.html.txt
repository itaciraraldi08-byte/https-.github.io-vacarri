const bulbs = document.getElementById("bulbs");
const total = 72;

for (let i = 0; i < total; i++) {
  const b = document.createElement("div");
  b.className = "bulb";

  const angle = (360 / total) * i;
  const rx = 1300;
  const ry = 600;

  const x = 1400 + Math.cos(angle * Math.PI / 180) * rx;
  const y = 700 + Math.sin(angle * Math.PI / 180) * ry;

  b.style.left = x + "px";
  b.style.top = y + "px";
  b.style.animationDelay = (i * 0.05) + "s";

  bulbs.appendChild(b);
}
